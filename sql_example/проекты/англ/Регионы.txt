SELECT DISTINCT
CASE 
        WHEN ci.city IN ('Алматы', 'Almaty') THEN 'Almay city'
        WHEN ci.city IN ('Астана', 'Астана (Нур-Султан)', 'Nur-Sultan', 'Astana') THEN 'Astana city'
        WHEN ci.city IN ('Шымкент', 'Shymkent') THEN 'Shymkent city'
        WHEN ci.city IN ('Семей', 'Semey') THEN 'Abai region'
        WHEN ci.city IN ('Жезказган', 'Zhezkazgan') THEN 'Ulytau region'
        WHEN ci.city IN ('Талдыкорган', 'Taldykorgan') THEN 'Zhetisu region'
        ELSE s.name_en
    END AS 'Регион заявителя',

FROM applicants a 
LEFT JOIN states s ON a.state_id = s.id
LEFT join cities ci ON ci.id=a.city_id 
WHERE s.country_id=222 AND s.name is NOT null
