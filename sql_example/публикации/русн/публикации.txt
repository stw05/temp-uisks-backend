set sql_mode='';

SELECT
  o.id,
  o.number,
  ct.name,
  pr.name_ru,
  ot.name_ru,
  a.name,
  d.name_ru,
  SUM(iic.scopus_publications_count)        AS scopus_total,
  SUM(iic.webofscience_publications_count)  AS wos_total,
  SUM(iic.publication_native)               AS publication_native_total,
  SUM(iic.publication_foreign)              AS publication_foreign_total,
  sum(iic.patents_count) AS patents_total,
  sum(iic.app_count) AS bnde_total
  
FROM obj_main o
LEFT JOIN obj_main o1      ON o.id = o1.parent_id
LEFT JOIN ic_info_card iic ON o1.id = iic.obj_id
LEFT JOIN competition_types ct ON o.comp_type_id = ct.id
LEFT JOIN obj_priorities op ON o.id = op.obj_id
LEFT JOIN priorities pr ON pr.id=op.priority_id
LEFT join overall_types ot ON ot.id=o.ov_type_id
LEFT JOIN applicants a ON a.id=o.applicant_id
LEFT JOIN departament d ON d.id=o.departament_id
LEFT JOIN states s ON a.state_id = s.id
LEFT join cities ci ON ci.id=a.city_id
LEFT join obj_mrnti om ON o.id = om.obj_id

WHERE o.status=11 AND o.obj_type_id not IN (3,5)
/*WHERE o.number='AP19178659'*/
GROUP BY o.id;
